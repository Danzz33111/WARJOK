  <script>
    const filtersContainer = document.getElementById("filters");
    const music = document.getElementById("bgMusic");
    let isMuted = false;

    // Filter
    document.querySelectorAll('.filter-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
        const filter = btn.dataset.filter;
        document.querySelectorAll('.photo-card').forEach(card => {
          card.classList.toggle('hidden', filter !== 'all' && !card.classList.contains(filter));
        });
      });
    });

    // Canvas Stars & Meteor (ringkas)
    const canvas = document.getElementById("stars-canvas");
    const ctx = canvas.getContext("2d", { alpha: true });
    let w = canvas.width = window.innerWidth;
    let h = canvas.height = window.innerHeight;
    let layers = [], meteors = [], time = 0, scrollY = 0;

    function initLayers() { /* ... (sama seperti sebelumnya) */ 
      const isMobile = w < 768;
      layers = [{ count: isMobile ? 130 : 230, speed: 0.015, size: 1.1, opacity: 0.95 },
                { count: isMobile ? 85 : 165, speed: 0.04, size: 1.6, opacity: 0.75 },
                { count: isMobile ? 55 : 105, speed: 0.085, size: 2.2, opacity: 0.55 }];
      layers.forEach(layer => layer.stars = Array.from({length: layer.count}, () => ({
        x: Math.random()*w, y: Math.random()*h, baseOpacity: Math.random()*0.65+0.45,
        twinklePhase: Math.random()*Math.PI*2, size: layer.size
      })));
    }

    function drawStars() { /* ... sama ... */ ctx.clearRect(0,0,w,h); /* stars drawing */ }
    function createMeteor() { if (Math.random() < 0.022) { /* ... */ } }
    function drawMeteors() { /* ... sama ... */ }
    function animate() { time += 0.018; drawStars(); createMeteor(); drawMeteors(); requestAnimationFrame(animate); }

    // Lightbox
    let currentIndex = 0;
    let autoSlideEnabled = false;
    let slideInterval = null;

    const photos = document.querySelectorAll(".photo-card");
    const lightbox = document.getElementById("lightbox");
    const mediaContainer = document.getElementById("media-container");
    const captionEl = document.getElementById("lightbox-caption");
    const autoBtn = document.getElementById("auto-slide-btn");
    const volumeBtn = document.getElementById("volume-btn");

    function showSlide() {
      mediaContainer.innerHTML = "";
      const card = photos[currentIndex];
      const media = card.querySelector("img, video");

      if (media.tagName === "VIDEO") {
        const v = document.createElement("video");
        v.src = media.src; v.controls = true; v.autoplay = true; v.playsInline = true; v.loop = true;
        mediaContainer.appendChild(v);
      } else {
        const img = document.createElement("img"); img.src = media.src; mediaContainer.appendChild(img);
      }
      captionEl.textContent = card.querySelector(".caption").textContent || "";
    }

    function goToNext() { currentIndex = (currentIndex + 1) % photos.length; showSlide(); }
    function goToPrev() { currentIndex = (currentIndex - 1 + photos.length) % photos.length; showSlide(); }

    function toggleAutoSlide() {
      autoSlideEnabled = !autoSlideEnabled;
      autoBtn.textContent = autoSlideEnabled ? "Auto Slide ON" : "Auto Slide OFF";
      autoBtn.classList.toggle("active", autoSlideEnabled);
      clearInterval(slideInterval);
      if (autoSlideEnabled) slideInterval = setInterval(goToNext, 3200);
    }

    function toggleVolume() {
      isMuted = !isMuted;
      music.muted = isMuted;
      volumeBtn.textContent = isMuted ? "🔇" : "🔊";
    }

    function toggleMinimize() { lightbox.classList.toggle("minimized"); }

    function closeLightbox() {
      lightbox.classList.remove("active", "minimized");
      clearInterval(slideInterval);
      autoSlideEnabled = false;
      autoBtn.textContent = "Auto Slide OFF";
      autoBtn.classList.remove("active");
      const v = mediaContainer.querySelector("video");
      if (v) v.pause();
      filtersContainer.classList.remove("hidden");
    }

    photos.forEach((card, i) => {
      card.addEventListener("click", () => {
        currentIndex = i;
        lightbox.classList.add("active");
        showSlide();
        filtersContainer.classList.add("hidden");
      });
    });

    // Music
    const overlay = document.getElementById("overlay");
    overlay.addEventListener("click", () => {
      overlay.classList.add("fade-out");
      setTimeout(() => overlay.style.display = "none", 1000);
      music.volume = 0.42;
      music.play();
    });

    window.addEventListener("resize", () => { w = canvas.width = window.innerWidth; h = canvas.height = window.innerHeight; initLayers(); });
    window.addEventListener('scroll', () => { scrollY = window.scrollY; });

    initLayers();
    animate();
  </script>
